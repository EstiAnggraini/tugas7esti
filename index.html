🎯 Pengantar Generic

Apa itu Generic? Fitur untuk membuat kode yang bekerja dengan berbagai tipe data.

Analog: Kotak penyimpanan yang bisa menampung apa saja, tapi dengan label tipe.

📌 Manfaat Generic:
✅ Type safety (keamanan tipe)
✅ Code reusability (penggunaan ulang)
✅ Compile-time checking

📝 Sintaks Generic Dasar

Type parameter: <T>, <E>, <K, V>
Konvensi:
• T = Type
• E = Element (collection)
• K = Key, V = Value
• R = Return type
✅ Contoh Benar
// Nama: Azriel Miracle Yuandre
// NIM: 251410024
// KELAS: SI2A         
class MyClass<T> {
  T data;
  MyClass(this.data);
  T getData() => data;
}

T myFunc<T>(T value) => value;

void main() {
  // Class generic
  var myIntClass = MyClass<int>(10);
  var myStringClass = MyClass<String>("test");
  print(myIntClass.getData());   // Output: 10
  print(myStringClass.getData()); // Output: test

  // Fungsi generic
  var intValue = myFunc<int>(100);
  var stringValue = myFunc<String>("hello");
  print(intValue);   // Output: 100
  print(stringValue); // Output: hello
}

class Box<T> {
  T? _value;
  Box(this._value);
  T? get value => _value;
  void setValue(T value) { _value = value; }
}

void main() {
  // Box dengan tipe int
  var intBox = Box<int>(10);
  print('Int Box: ${intBox.value}'); // Output: Int Box: 10
  intBox.setValue(20);
  print('After set: ${intBox.value}'); // Output: After set: 20

  // Box dengan tipe String
  var stringBox = Box<String>("Hello, Generic!");
  print('String Box: ${stringBox.value}'); // Output: String Box: Hello, Generic!
  stringBox.setValue('Hello World');
  print('After set: ${stringBox.value}'); // Output: After set: Hello World

  // Box dengan tipe double
  var doubleBox = Box<double>(3.14);
  print('Double Box: ${doubleBox.value}'); // Output: Double Box: 3.14
}

🔒 Generic dengan Extends (Constraints)

📌 Keyword extends → membatasi tipe data yang dapat digunakan.
✅ Contoh Benar

  T value;
  NumberBox(this.value);
  T getValue() => value;
}

void main() {
  // ✅ Valid - int dan double adalah turunan num
  var intBox = NumberBox<int>(10);
  print("int box: ${intBox.getValue()}"); // Output: int box: 10

  var doubleBox = NumberBox<double>(3.14);
  print("double box: ${doubleBox.getValue()}"); // Output: double box: 3.14
}
❌ Contoh ERROR
// ❌ ERROR: String bukan turunan dari num

// 1. Class Repository
class Repository {
  final Map _storage = {};
  int _nextId = 1;

  int save(T item) {
    int id = _nextId++;
    _storage[id] = item;
    print('✅ Saved: ID $id → $item');
    return id;
  }

  T? getById(int id) {
    return _storage[id];
  }

  List getAll() {
    return _storage.values.toList();
  }

  void printAll() {
    if (_storage.isEmpty) {
      print('📭 Repository kosong');
      return;
    }
    print('📦 Isi Repository:');
    _storage.forEach((id, item) => print('   ID $id: $item'));
  }
}

// 2. Fungsi swap
void swap(List list, int i, int j) {
  if (i < 0 || i >= list.length || j < 0 || j >= list.length) {
    print('❌ Error: Indeks tidak valid!');
    return;
  }
  if (i == j) return;
  
  T temp = list[i];
  list[i] = list[j];
  list[j] = temp;
}

// 3. Class Triple
class Triple {
  A first;
  B second;
  C third;

  Triple(this.first, this.second, this.third);

  A getFirst() => first;
  B getSecond() => second;
  C getThird() => third;

  void setFirst(A value) => first = value;
  void setSecond(B value) => second = value;
  void setThird(C value) => third = value;

  void printTriple() {
    print('($first, $second, $third)');
  }

  List toList() => [first, second, third];
}

🏆 Challenge: Sistem Repository Generic

✅ Solusi
// Nama: Azriel Miracle Yuandre
// NIM: 251410024
// KELAS: SI2A           
class Product {
  final int id;
  final String name;
  final double price;
  Product(this.id, this.name, this.price);
  @override
  String toString() => 'Product($id, $name, $price)';
}

class User {
  final int id;
  final String username;
  User(this.id, this.username);
  @override
  String toString() => 'User($id, $username)';
}

class GenericRepository<T> {
  final Map<int, T> _items = {};
  int _nextId = 1;

  T? create(T item) {
    _items[_nextId] = item;
    return item;
  }

  T? read(int id) {
    return _items[id];
  }

  T? update(int id, T item) {
    if (_items.containsKey(id)) {
      _items[id] = item;
      return item;
    }
    return null;
  }

  bool delete(int id) {
    return _items.remove(id) != null;
  }

  List<T> readAll() {
    return _items.values.toList();
  }

  void printAll() {
    _items.forEach((id, item) => print('ID $id: $item'));
  }
}

void main() {
  var productRepo = GenericRepository<Product>();
  productRepo.create(Product(1, 'Laptop', 15000000));
  productRepo.create(Product(2, 'Mouse', 250000));
  print('Products: ${productRepo.readAll()}');

  var userRepo = GenericRepository<User>();
  userRepo.create(User(1, 'PRABOWO'));
  userRepo.create(User(2, 'GIBRAN'));
  print('Users: ${userRepo.readAll()}');
}

📖 Ringkasan

🎯 Poin Penting:
1. Generic membuat kode type-safe dan reusable
2. Bisa digunakan di class, fungsi, collections
3. Type inference memudahkan penggunaan
4. Constraints (extends) membatasi tipe yang diperbolehkan
📌 Best Practice:
• Gunakan generic untuk menghindari dynamic
• Gunakan extends untuk membatasi tipe
• Manfaatkan type inference untuk kode lebih bersih
